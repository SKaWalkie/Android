package com.example.myapplication;

import android.database.Cursor;
import android.database.sqlite.SQLiteDatabase;
import android.os.Bundle;
import android.view.View;
import android.widget.EditText;
import android.widget.TextView;
import androidx.appcompat.app.AppCompatActivity;

public class MainActivity extends AppCompatActivity {

    SQLiteDatabase db;

    @Override
    protected void onCreate(Bundle b) {
        super.onCreate(b);
        setContentView(R.layout.activity_main);

        // 1. Create or open the database & table in one line!
        db = openOrCreateDatabase("CustomerDB", MODE_PRIVATE, null);
        db.execSQL("CREATE TABLE IF NOT EXISTS Customer(id INTEGER PRIMARY KEY AUTOINCREMENT, name TEXT, address TEXT, phone TEXT)");
    }

    // 2. Insert new customer details
    public void insert(View v) {
        db.execSQL("INSERT INTO Customer(name, address, phone) VALUES('" + getTxt(R.id.etName) + "','" + getTxt(R.id.etAddress) + "','" + getTxt(R.id.etPhone) + "')");
    }

    // 3. Display all customer details
    public void display(View v) {
        Cursor c = db.rawQuery("SELECT * FROM Customer", null);
        String result = "";

        // Loop through the database results row by row
        while (c.moveToNext()) {
            result += "ID: " + c.getString(0) + "\nName: " + c.getString(1) +
                    "\nAddress: " + c.getString(2) + "\nPhone: " + c.getString(3) + "\n\n";
        }
        ((TextView) findViewById(R.id.tvResult)).setText(result);
    }

    // Quick helper method to keep the insert code tiny
    private String getTxt(int id) {
        return ((EditText) findViewById(id)).getText().toString();
    }
}

import android.app.AlertDialog;
import android.database.Cursor;
import android.database.sqlite.SQLiteDatabase;
import android.os.Bundle;
import android.view.View;
import android.widget.EditText;
import android.widget.Toast;
import androidx.appcompat.app.AppCompatActivity;

public class MainActivity extends AppCompatActivity {
    SQLiteDatabase db;
    EditText u, p;

    @Override protected void onCreate(Bundle b) {
        super.onCreate(b);
        setContentView(R.layout.activity_main);
        u = findViewById(R.id.u); p = findViewById(R.id.p);

        // 1. Create DB and Table in just two lines
        db = openOrCreateDatabase("LoginDB", MODE_PRIVATE, null);
        db.execSQL("CREATE TABLE IF NOT EXISTS Users(user TEXT, pass TEXT)");
        
        // Insert a dummy user so we can actually test the login
        Cursor c = db.rawQuery("SELECT * FROM Users", null);
        if (c.getCount() == 0) {
            db.execSQL("INSERT INTO Users VALUES('admin', '123')");
        }
        c.close();
    }

    public void login(View v) {
        String userStr = u.getText().toString();
        String passStr = p.getText().toString();

        // 2. Query the database to check if the username and password match
        Cursor c = db.rawQuery("SELECT * FROM Users WHERE user=? AND pass=?", new String[]{userStr, passStr});

        if (c.getCount() > 0) {
            // 3. Show Alert Dialog on Success
            new AlertDialog.Builder(this)
                .setTitle("Success")
                .setMessage("Login Successful")
                .setPositiveButton("OK", null)
                .show();
        } else {
            Toast.makeText(this, "Invalid Username or Password", Toast.LENGTH_SHORT).show();
        }
        c.close();
    }
}

import android.os.Bundle;
import android.view.View;
import android.widget.Button;
import android.widget.EditText;
import androidx.appcompat.app.AppCompatActivity;

public class MainActivity extends AppCompatActivity {
    EditText disp;
    double num1 = 0;
    String op = "";

    @Override protected void onCreate(Bundle b) {
        super.onCreate(b);
        setContentView(R.layout.activity_main);
        disp = findViewById(R.id.disp);
    }

    public void calc(View v) {
        // Get the text of the button that was clicked
        String b = ((Button) v).getText().toString();
        String currentText = disp.getText().toString();

        if (b.matches("[0-9.]")) { 
            // If it's a number or decimal, append it
            disp.append(b);
            
        } else if (b.equals("C")) { 
            // Clear screen
            disp.setText("");
            
        } else if (b.equals("=")) { 
            // Calculate Result
            double num2 = Double.parseDouble(currentText);
            double res = op.equals("+") ? num1 + num2 : 
                         op.equals("-") ? num1 - num2 : 
                         op.equals("*") ? num1 * num2 : num1 / num2;
                         
            disp.setText(String.valueOf(res));
            
        } else { 
            // It's an operator (+, -, *, /)
            num1 = Double.parseDouble(currentText);
            op = b;
            disp.setText(""); // Clear for the next number
        }
    }
}

package com.example.myapplication;

import android.os.Bundle;
import android.view.View;
import android.view.ViewGroup;
import android.widget.CheckBox;
import android.widget.TextView;
import androidx.appcompat.app.AppCompatActivity;

public class MainActivity extends AppCompatActivity {

    @Override protected void onCreate(Bundle b) { super.onCreate(b); setContentView(R.layout.activity_main); }

    // Linked directly to the button in XML
    public void show(View v) {
        String selected = "Selected Hobbies:\n";
        ViewGroup group = findViewById(R.id.cbGroup);

        // Loop through the checkbox group
        for (int i = 0; i < group.getChildCount(); i++) {
            CheckBox cb = (CheckBox) group.getChildAt(i);
            if (cb.isChecked()) selected += "- " + cb.getText() + "\n";
        }

        // Display the result
        ((TextView) findViewById(R.id.result)).setText(selected);
    }
}

        sp.setOnItemSelectedListener(new AdapterView.OnItemSelectedListener() {
            @Override public void onItemSelected(AdapterView<?> p, View v, int i, long id) {

    }
}

java
package com.example.myapplication;

import android.os.Bundle;
import android.view.View;
import android.widget.EditText;
import android.widget.TextView;
import android.widget.Toast;
import androidx.appcompat.app.AppCompatActivity;

public class MainActivity extends AppCompatActivity {
    EditText e1, e2; TextView tv;

    @Override protected void onCreate(Bundle b) {
        super.onCreate(b);
        setContentView(R.layout.activity_main);
        e1 = findViewById(R.id.n1); e2 = findViewById(R.id.n2); tv = findViewById(R.id.res);
    }

    public void check(View v) {
        int a = Integer.parseInt(e1.getText().toString());
        int b = Integer.parseInt(e2.getText().toString());

        if (a > 10 && b > 10) {
            // Rejection Logic
            Toast.makeText(this, "Both > 10! Enter new numbers.", Toast.LENGTH_LONG).show();
            e1.setText(""); e2.setText(""); // Clear inputs
            tv.setText("");
        } else {
            // Acceptance Logic
            tv.setText("Numbers: " + a + " and " + b);
        }
    }
}
